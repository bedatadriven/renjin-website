55 dens = dgamma(1./x, n/2, 2./n, give_log);
56 return give_log ? dens - 2*log(x): dens/(x*x);

60 q = n*f;
61 p = x*m*f;
62
63 if (m >= 2) {
64 f = m*q/2;
65 dens = dbinom_raw((m-2)/2, (m+n-2)/2, p, q, give_log);
66 }
67 else {
68 f = m*m*q / (2*p*(m+n));
69 dens = dbinom_raw(m/2, (m+n)/2, p, q, give_log);
70 }
71 return(give_log ? log(f)+dens : f*dens);
72}
